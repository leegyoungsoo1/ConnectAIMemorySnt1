<create_file path="read_weather.py">
import json
import os

def main():
    file_path = 'weather_data.json'
    
    if not os.path.exists(file_path):
        print(f"Error: {file_path} not found.")
        return

    try:
        with open(file_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
            
        print(f"{'City':<10} | {'Temp':<7} | {'Condition':<10} | {'Humidity':<10}")
        print("-" * 45)
        for entry in data:
            print(f"{entry['city']:<10} | {entry['temperature']:<7} | {entry['condition']:<10} | {entry['humidity']}%")
            
    except json.JSONDecodeError:
        print("Error: Failed to decode JSON.")
    except Exception as e:
        print(f"An unexpected error occurred: {e}")

if __name__ == "__main__":
    main()

        async function loadWeatherData() {
            const contentDiv = document.getElementById('weather-content');
            try {
                // 준혁님이 만든 weather_data.json 파일 fetch
                const response = await fetch('weather_data.json');
                if (!response.ok) throw new Error('파일을 찾을 수 없습니다.');
                
                const data = await response.json();
                
                // 데이터가 배열인 경우 첫 번째 요소 사용, 객체인 경우 그대로 사용

                contentDiv.innerHTML = `
                    <div class="info-item">

                    </div>

                        <span class="value">${weather.temperature || '-'}°C</span>
                    </div>
                    <div class="info-item">
                        <span class="label">☁️ 상태</span>

                        <span class="value">${weather.humidity || '-'}%</span>
                    </div>

            }
        }

        window.onload = loadWeatherData;
